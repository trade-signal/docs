## 新闻接口

Signal Feed 提供多来源的新闻资讯接口，支持新浪、富途、财联社、百度等主流财经新闻源。

## 基础信息

- **Base URL**: `http://localhost:4000/api/v1`
- **认证方式**: Bearer Token
- **数据格式**: JSON
- **接口前缀**: `/news`

## 接口列表

### 1. 获取新闻列表

获取多来源的新闻列表，支持指定新闻源。

```http
GET /api/v1/news/list?source=sina&page=1&pageSize=20
```

**查询参数**:

| 参数 | 类型 | 必需 | 默认值 | 说明 |
|------|------|------|--------|------|
| `source` | string | 是 | sina | 新闻来源：sina(新浪)、futunn(富途)、cls(财联社)、baidu(百度) |
| `page` | number | 否 | 1 | 页码 |
| `pageSize` | number | 否 | 20 | 每页数量 |

## Node.js 使用示例

```javascript
const API_TOKEN = 'your_api_token';
const BASE_URL = process.env.API_BASE_URL || 'http://localhost:4000/api/v1';

// 基础请求函数
async function apiRequest(endpoint, options = {}) {
  const response = await fetch(`${BASE_URL}${endpoint}`, {
    headers: {
      'Authorization': `Bearer ${API_TOKEN}`,
      'Content-Type': 'application/json',
      ...options.headers
    },
    ...options
  });
  
  if (!response.ok) {
    const error = await response.json();
    throw new Error(`API 错误: ${error.message} (${error.code})`);
  }
  
  return response.json();
}

// 获取多来源新闻列表
async function getNewsList(source = 'sina', page = 1, pageSize = 20) {
  const params = new URLSearchParams({
    source,
    page: page.toString(),
    pageSize: pageSize.toString()
  });
  
  return apiRequest(`/news/list?${params}`);
}

// 获取新浪新闻
async function getSinaNews(page = 1, pageSize = 20) {
  const params = new URLSearchParams({
    page: page.toString(),
    pageSize: pageSize.toString()
  });
  
  return apiRequest(`/news/sina/list?${params}`);
}

// 获取富途新闻
async function getFutunnNews(page = 1, pageSize = 20) {
  const params = new URLSearchParams({
    page: page.toString(),
    pageSize: pageSize.toString()
  });
  
  return apiRequest(`/news/futunn/list?${params}`);
}

// 获取财联社新闻
async function getClsNews(page = 1, pageSize = 20) {
  const params = new URLSearchParams({
    page: page.toString(),
    pageSize: pageSize.toString()
  });
  
  return apiRequest(`/news/cls/list?${params}`);
}

// 获取百度新闻
async function getBaiduNews(page = 1, pageSize = 20) {
  const params = new URLSearchParams({
    page: page.toString(),
    pageSize: pageSize.toString()
  });
  
  return apiRequest(`/news/baidu/list?${params}`);
}

// 使用示例
async function example() {
  try {
    // 获取新浪新闻
    const sinaNews = await getSinaNews(1, 10);
    console.log('新浪新闻:', sinaNews);
    
    // 获取富途新闻
    const futunnNews = await getFutunnNews(1, 10);
    console.log('富途新闻:', futunnNews);
    
    // 获取多来源新闻
    const allNews = await getNewsList('cls', 1, 10);
    console.log('财联社新闻:', allNews);
  } catch (error) {
    console.error('API 请求失败:', error);
  }
}
```

## 常见问题

### Q: 数据更新频率？
A: 新闻数据实时更新，各新闻源更新频率略有不同，基本每 30 分钟更新一次。

### Q: 支持哪些新闻来源？
A: 支持新浪(sina)、富途(futunn)、财联社(cls)、百度(baidu) 四个主要新闻源。

## 错误处理

```javascript
// 错误处理示例
async function handleApiError(response) {
  if (!response.ok) {
    const error = await response.json();
    throw new Error(`API 错误: ${error.message} (${error.code})`);
  }
  return response.json();
}

// 使用重试机制
async function apiRequestWithRetry(endpoint, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(`${BASE_URL}${endpoint}`, {
        headers: {
          'Authorization': `Bearer ${API_TOKEN}`
        }
      });
      return await handleApiError(response);
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)));
    }
  }
}
```
